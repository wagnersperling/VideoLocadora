
create database DBVideoLocadora

USE DBVideoLocadora 

CREATE TABLE GENRETB (
    IDGENRE INT PRIMARY KEY NOT NULL IDENTITY,
    NAMEGENRE VARCHAR (50),
    CREATIONDATE DATE,
    ACTIVO BIT
)

CREATE TABLE MOVIETB (
    IDMOVIE INT PRIMARY KEY NOT NULL IDENTITY,
    NAMEMOVIE VARCHAR (50),
    CREATIONDATE DATE,
    ACTIVO BIT,
    IDGENRE INT FOREIGN KEY REFERENCES GENRETB(IDGENRE)
)

CREATE TABLE LOCATIONTB (
    IDLOCATION INT PRIMARY KEY NOT NULL IDENTITY,
    CPF VARCHAR (15),
    CREATIONDATE DATE
)

CREATE TABLE LOGINTB (
    IDLOGIN INT PRIMARY KEY NOT NULL IDENTITY,
    USEREMAIL VARCHAR (50),
    USERPASSWORD VARCHAR (50)
)

CREATE TABLE MOVIEXLOCATION (
    ID INT PRIMARY KEY NOT NULL IDENTITY,
    IDLOCATION INT FOREIGN KEY REFERENCES LOCATIONTB(IDLOCATION),
    IDMOVIE INT FOREIGN KEY REFERENCES MOVIETB(IDMOVIE)
)
